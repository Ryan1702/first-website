# 链表实现栈

``` c++
#include<stdio.h>
#include<stdlib.h>
typedef struct Node
{
    int data;
    struct Node* next;
}Node;
int isEmpty(Node* top)
{
    return (top==NULL);
}
void push(Node** top,int value)
{
    Node* newnode=(Node*)malloc(sizeof(Node));
    newnode->data=value;
    newnode->next=*top;
    *top=newnode;
}
void pop(Node** top)
{
    if(isEmpty(*top)) return ;
    Node* temp=*top;
    *top=temp->next;
    free(temp);
}
void peek(Node* top)
{
    if(isEmpty(top)) printf("Stack is empty\n");
    else printf("Top element is: %d\n",top->data);
}
void display(Node* top)
{
    if(isEmpty(top)) printf("Stack is empty\n");
    Node* temp=top;
    printf("Stack elements:\n");
    while(temp!=NULL)
    {
        printf("%d ",temp->data);
        temp=temp->next;
    }
    printf("\n");
}
int main()
{
    return 0;
}
```
