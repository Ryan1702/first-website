# 数组实现栈

``` c++
#include<stdio.h>
#define MAX 1000
int stack[MAX],top=-1;
void push(int value)
{
    if(top==MAX-1) printf("Stack Overflow\n");
    else stack[++top]=value;
}
void pop()
{
    if(top==-1) printf("Stack Underflow\n");
    else printf("Popped element: %d\n",stack[top--]);
}
int main()
{
    return 0;
}
```
